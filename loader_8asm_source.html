15gdt_start: ; Beginning of GDT-table. The table itself contains information about RAM segments.
16 ; Each segment has a set of parameters (e.g. readable/writable, data/code, etc.)
17 ; This table is a basic flat model table with no security required to go to 32-bit CPU mode.
18 ; The table contains 2 overlapping 4GB segments (code and data). Since they overlap, they're not protected.
19 ; It maybe will be changed later, from protected mode along with other interesting features.
20
21gdt_null: ; First sector has to be null. For catching segment-register-not-set error catching.
22 dd 0x0 ; Info about one segment takes 4 words (8 bytes) of memory.
23 dd 0x0 ; This null-segment is apparently initialized with all zeroes.
24
25gdt_code: ; The code segment.
26 dw 0xffff ; Segment limit - how long is it. 4GB, as stated.
27 dw 0x0000 ; Segment base - where it begins in memory. In the very beginning.
28 db 0x00 ; Segment base one more time (for some reason it has to be initialized two times).
29 db 10011010b ; Segment flags, these are:
30 ; Present (1) - the segment is present.
31 ; Privilege (00) - highest privilege.
32 ; Descriptor type (1) - 1 for code and data, 0 for traps.
33 ; Type flags:
34 ; Type (1) - 1 for code, 0 for data.
35 ; Conforming (0) - means lower-privileged segments can't call it.
36 ; Readable (1) - readable or executable-only.
37 ; Accessed (0) - CPU sets when accesses the segment.
38 db 11001111b ; Segment flags:
39 ; Granularity (1) - Expands limit from 0xffff to 0xffff000.
40 ; 32 bit (1) - set 0 for 64-bit code.
41 ; 64 bit (0) - we have it unused.
42 ; AVL (0) - flag for custom uses.
43 ; Other 4 bits (1111) contain info about segment limit.
44 db 0x00 ; Segment base (for the third time, yes).
45
46gdt_data: ; The data segment (I will comment differences only).
47 dw 0xffff
48 dw 0x0000
49 db 0x00
50 db 10010010b ; Type (0) - for data.
51 db 11001111b
52 db 0x00

65read_disk: ; Procedure for reading sectors from disc into memory.
66 ; DH = how many sectors; DL = from what drive; BX = where to load.
67 pusha ; Push all registers.
68 push dx ; For later error checking.
69
70 mov ah, 0x02 ; int 13h : 02 - reading from disk BIOS function.
71 mov al, dh ; Sectors number.
72 mov ch, 0x00 ; Track cylinder.
73 mov dh, 0x00 ; Track number.
74 mov cl, 0x02 ; First sector to read.
75
76 int 13h
77
78 jc .disk_error ; int 13h sets -c flag if reading error occurs.
79
80 pop dx ; int 13h stores number of sectors read into al.
81 cmp dh, al ; Checking how many sectors were read vs. how many must've been read.
82 jne .disk_error ; If not the same number handling error.
83 jmp .done
84
85.disk_error:
86 mov si, ERROR_MSG ; Setting and printing error message.
87 call write_string
88 stc ; Ensuring -c flag was set to check if function finished successfully.
89
90.done:
91 popa ; Pop all registers.
92 ret

95write_string: ; Procedure for writing string into console.
96 ; SI = pointer to string in ASCIIZ format.
97 pusha
98 mov ah, 0x0e ; int 10h : 0e - writing char into console and setting cursor to next position BIOS function.
99 ; Other parameters (color, mode, etc.) are set by default.
100
101.repeat:
102 lodsb ; Loading char from SI to AL, increasing SI.
103 cmp al, 0x00 ; Checking if the end of string was reached.
104 je .done ; Escaping cycle.
105 int 10h
106 jmp .repeat ; Proceeding to next char.
107
108.done:
109 popa
110 ret

113kernel_start: ; Start of booting code.
114 mov [BOOT_DRIVE], dl ; BIOS puts number of booting device into DL. Saving it into variable.

116 cli ; Clearing interruptions before SP and SP modifying.
117 mov ax, cs ; CS points to 7c00 at the boot time, the current segment. TODO: check!!
118 mov ss, ax ; Placing stack to the same segment. TODO: So stack will be from 0x7c00 to 0x500, check that's enough!
119 mov bp, start ; Stack grows downwards in memory, while the code - upwards. TODO: check!!
120 mov sp, bp ; So stack starts right in the entry point, but grows to another direction.
121 sti ; Setting interruptions back.

123 mov ds, ax ; Setting DS and ES to the same segment.
124 mov es, ax ; My boot loader contains code and variables all together.
