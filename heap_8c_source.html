45/**
46 * Internal function for space allocation in heap.
47 * Sets up and fills heap block header, optionally alters headers of the next and previous heap blocks.
48 * @param free_pointer pointer to free space (heap block header size + required size).
49 * @param size required size of heap block.
50 * @param previous previous heap block.
51 * @param next next heap block.
52 * @return pointer to the start of the block.
53 */
54static void* allocate_space(void* free_pointer, u_dword size, heap_block_header* previous, heap_block_header* next) {
55 ((heap_block_header*) free_pointer)->size = size;
56 ((heap_block_header*) free_pointer)->previous = previous;
57 ((heap_block_header*) free_pointer)->next = next;
58
59 void* memory_address = free_pointer + sizeof(heap_block_header);
60 if (previous) get_header(previous)->next = memory_address;
61 if (next) get_header(next)->previous = memory_address;
62 return memory_address;
63}

69 * This implementation of heap has fixed size, and so doesn't rely on paging.
70 * That's enough for kernel and basic app needs, TODO: provide another heap implementation for extended MushLib.

84 * Searches for a place of suitable size to fit a structure of requested size in.
85 * Throws heap exception if no free space is available.

136/**
137 * Function for heap block size alteration.
138 * Attempts to alter size in-place and copys the block only if in-place growth is not available.
139 * @param structure pointer to the structure for reallocation.
140 * @param new_size new requested size of the structure.
141 * @return pointer to the new structure.
142 */
143void* realloc(void* structure, u_dword new_size) {
144 heap_block_header* block = get_header(structure);
145 if (new_size < block->size) {
146 block->size = new_size;
147 } else if (new_size > block->size) {
148 if (new_size < block->next - structure) block->size = new_size;
149 else {
150 void* new_structure = malloc(new_size);
151 memory_copy(structure, new_structure, block->size);
152 free(structure);
153 structure = new_structure;
154 }
155 }
156 return structure;
157}
