219static page_pointer* get_page(u_dword address, u_dword make_tail, page_directory *dir) {
220 if (address > memory_end) return nullptr;
221 // Turn the address into an index.
222 address /= page_size;
223 // Find the page table containing this address.
224 u_dword table_idx = address / 1024;
225 if (get_page_table_pointer(dir->tablesPhysical[table_idx])) { // If this table is already assigned
226 return &get_page_table_pointer(dir->tablesPhysical[table_idx])->pages[address % 1024];
227 } else if (make_tail) {
228 page_table* page_table_pointer = nullptr;
229 if ((address < orbit_start) || (address >= memory_end))
230 page_table_pointer = (page_table*) k_malloc_aligned(sizeof(page_table), true);
231 else page_table_pointer = allocate_page_table();
232 dir->tablesPhysical[table_idx] = create_page_table_entry(page_table_pointer, make_tail);
233 memory_clear((byte*) page_table_pointer, sizeof(page_table), 0);
234 return &get_page_table_pointer(dir->tablesPhysical[table_idx])->pages[address % 1024];
235 } else return nullptr;
236}

248void initialise_paging() {
249 // The size of physical memory. For the moment we
250 // assume it is 16MB big.
251 u_dword page_pool_size = (memory_end - orbit_start) / page_size / 8;
252 page_pool = k_malloc_aligned(page_pool_size, false);
253 memory_clear((byte*) page_pool, page_pool_size, 0);
254
255 u_dword page_table_pool_size = (page_dirs - page_tables) / sizeof(page_table) / 8;
256 page_pool = k_malloc_aligned(page_table_pool_size, false);
257 memory_clear((byte*) page_pool, page_table_pool_size, 0);
258
259 // Let's make a page directory.
260 kernel_directory = (page_directory*) k_malloc_aligned(sizeof(page_directory), true);
261 memory_clear((byte*) kernel_directory, sizeof(page_directory), 0);
262
263 for (u_dword j = 0; j < orbit_start; j += page_size)
264 *get_page(j, 0x00000003, kernel_directory) = create_page_entry((const byte*) j, 0x00000003);
265 for (u_dword j = memory_end; j < memory_edge; j += page_size)
266 *get_page(j, 0x00000003, kernel_directory) = create_page_entry((const byte*) j, 0x00000003);
267
268 // Before we enable paging, we must register our page fault handler.
269 set_interrupt_handler(14, page_fault);
270
271 current_directory = kernel_directory;
272
273 // Now, enable paging!
274 asm volatile("mov %0, %%cr3":: "r"(&current_directory->tablesPhysical));
275 u_dword cr0, cr3;
276 asm volatile("mov %%cr3, %0": "=r"(cr3));
277 asm volatile("mov %%cr0, %0": "=r"(cr0));
278 cr0 |= 0x80000001; // Enable paging!
279 asm volatile("mov %0, %%cr0":: "r"(cr0));
280}

284void page_fault(registers* regs) {
285 // A page fault has occurred.
286 // The faulting address is stored in the CR2 register.
287 u_dword faulting_address;
288 asm ("mov %%cr2, %0" : "=r" (faulting_address));
289
290 // The error code gives us details of what happened.
291 int present = regs->err_code & 0x1; // Page not present
292 int rw = regs->err_code & 0x2; // Write operation?
293 int us = regs->err_code & 0x4; // Processor was in user-mode?
294 int reserved = regs->err_code & 0x8; // Overwritten CPU-reserved bits of page entry?
295 int id = regs->err_code & 0x10; // Caused by an instruction fetch?
296
297 // Output an error message.
298 bad("Page fault! ( ")
299 if (present) bad("present ")
300 if (!rw) bad("read-only ")
301 if (us) bad("user-mode ")
302 if (reserved) bad("reserved ")
303 bad(") at %h (EIP: %h)\n", faulting_address, regs->eip)
304 asm("jmp .");
305}
